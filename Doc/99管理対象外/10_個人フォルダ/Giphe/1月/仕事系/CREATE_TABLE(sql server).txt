create table spots (
  spot_id varchar(20) 
  , sgrid_x int not null 
  , sgrid_y int not null 
  , location_id int not null 
  , field_id int not null 
  , grid_x int default 1 not null 
  , grid_y int default 1 not null 
  , building_id int 
  , assets int 
  , civil_point int 
  , f_count int 
  , is_public tinyint 
  , version tinyint 
  , is_deleted tinyint 
  , ins_date datetime 
  , ins_id varchar(10) 
  , upd_date datetime 
  , upd_id varchar(10) 
  , primary key (sgrid_x,sgrid_y,grid_x,grid_y)
) 


create table areas (
  grid_x int default 1 not null 
  , grid_y int default 1 not null 
  , location_id int not null 
  , field_id int not null 
  , is_maxHeight tinyint not null 
  , height int not null 
  , is_river tinyint 
  , depth int 
  , landlord_gene_id int default 0 
  , character_id int default 0 
  , assets int default 0 not null 
  , civil_point int default 0 
  , family tinyint default 0 
  , is_public tinyint 
  , version tinyint 
  , is_deleted tinyint 
  , ins_date datetime 
  , ins_id varchar(10) 
  , upd_date datetime 
  , upd_id varchar(10) 
  , primary key (grid_x,grid_y)
)
